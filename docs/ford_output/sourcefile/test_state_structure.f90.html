function find_domain(state_indx) result(dom)
   integer(i8), intent(in) :: state_indx
   integer :: dom

   integer :: idom
   
   dom = -1
   do idom = 1, get_num_domains()
      if ( state_indx >= msize(idom) .and. &
           state_indx <= msize(idom+1) ) then
         dom = idom
         return
      endif
   enddo
         
end function find_domain
